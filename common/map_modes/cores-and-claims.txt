scripted_map_modes = {
	cores_and_claims_map_mode = {
		top = {
			type = state
			color = {
				FROM = {
					set_temp_variable = { alpha = 0.5 }
					if = {
						limit = { is_core_of = ROOT }
						set_temp_variable = { green = 0.85 }
						set_temp_variable = { blue = 1.0 }
					}
					else_if = {
						limit = { is_claimed_by = ROOT }
						set_temp_variable = { green = 0.8 }
					}
					else_if = {
						limit = { is_owned_by = ROOT }
						set_temp_variable = { red = 0.8 }
					}
					else = {
						set_temp_variable = { red = 0.35 }
						set_temp_variable = { green = 0.35 }
						set_temp_variable = { blue = 0.35 }
						set_temp_variable = { alpha = 0.25 }
					}
				}
			}
		}
 	
		bottom = {
			type = country
			color = {
				set_temp_variable = { red = 0.6 }
				set_temp_variable = { blue = 0.6 }
				set_temp_variable = { green = 0.6 }
				set_temp_variable = { alpha = 0.25 }
				
			}

			thickness = yes
		}
 
 
		# if we want to show texts on map this you can set far_text & near_text
		# must be one of the:
		#  none
		#  country
		#  state
		#  faction
		#  player
 	
		far_text = country
		near_text = state

		# if yes, the game will update the map mode daily_update
		# otherwise force_update_map_mode effect can be used
		update_daily = yes

	}
}
