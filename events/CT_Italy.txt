# Fight At Armory
country_event = {
	id = frencharmyinfluence.1
	title = frencharmyinfluence.1.t
	desc = frencharmyinfluence.1.d
	picture = GFX_report_event_armoryfight
	
	trigger = {
		has_country_flag = CT_French_Army_Presence_flag
		date > 1908.01.01
	}

	fire_only_once = yes
	
	option = {
		name = frencharmyinfluence.1.a
		ai_chance = { factor = 100 }
		ROOT = {
			add_to_variable = {
				var = CT_French_Army_variable
				value = 0.025
				tooltip = raise_FA_Influence_tt
			}
			send_equipment = {
				equipment = infantry_equipment
				amount = 100
				target = FRA
			}
			add_war_support = -0.05
		}
	}

	option = {
		name = frencharmyinfluence.1.b
		ai_chance = { factor = 100 }
		ROOT = {
			add_to_variable = {
				var = CT_French_Army_variable
				value = -0.025
				tooltip = lower_FA_Influence_tt
			}
		}
	}
}
